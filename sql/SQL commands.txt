
-- Create a table to store student information
CREATE TABLE STUDENT (
    STUDENT_ID INT PRIMARY KEY,              -- Unique identifier for each student (Primary Key)
    STUDENT_FNAME NVARCHAR(50),              -- Student's first name
    STUDENT_LNAME NVARCHAR(50),              -- Student's last name
    STUDENT_EMAIL NVARCHAR(100),             -- Student's email address
    STUDENT_PHONE NVARCHAR(15),              -- Student's phone number
    STUDENT_DOB DATE,                        -- Student's date of birth
    STUDENT_GENDER CHAR(1),                  -- Student's gender (e.g., 'M' for Male, 'F' for Female)
    STUDENT_ADDRESS NVARCHAR(255)            -- Student's residential address
);

-- Create a table to store instructor information
CREATE TABLE INSTRUCTOR (
    INSTR_ID INT PRIMARY KEY,                -- Unique identifier for each instructor (Primary Key)
    INSTR_NAME NVARCHAR(100),                -- Instructor's full name
    INSTR_EMAIL NVARCHAR(100),               -- Instructor's email address
    INSTR_PHONE NVARCHAR(15),                -- Instructor's phone number
    EXPERTISE_LEVEL NVARCHAR(50)             -- Instructor's level of expertise (e.g., Beginner, Intermediate, Expert)
);

-- Create a table to store course information
CREATE TABLE COURSE (
    CRS_CODE INT PRIMARY KEY,                -- Unique identifier for each course (Primary Key)
    CRS_NAME NVARCHAR(100),                  -- Name of the course
    CRS_DESCRIPTION NVARCHAR(255),           -- Description of the course
    CRS_TIME NVARCHAR(50),                   -- Scheduled time for the course (e.g., "10:00 AM - 12:00 PM")
    CRS_START_DATE DATE,                     -- Start date of the course
    CRS_END_DATE DATE,                       -- End date of the course
    LECTURE_LINK NVARCHAR(255),              -- Link to access course lectures (e.g., Zoom or Google Meet link)
    CRS_FEES FLOAT,                          -- Course fees
    INSTR_ID INT                             -- Foreign Key referencing the instructor teaching the course
);

-- Create a table to store student enrollment information
CREATE TABLE ENROLLMENT (
    STUDENT_ID INT,                          -- Foreign Key referencing the student enrolled
    CRS_CODE INT,                            -- Foreign Key referencing the course enrolled in
    ENROL_DATE DATE,                         -- Date of enrollment
    COMPLETION_STATE NVARCHAR(50),           -- Enrollment completion status (e.g., "In Progress", "Completed")
    GRADE NVARCHAR(2),                       -- Grade achieved by the student in the course
    PAYMENT NVARCHAR(50),                    -- Payment status (e.g., "Paid", "Pending")
    PRIMARY KEY (STUDENT_ID, CRS_CODE)       -- Composite Primary Key (STUDENT_ID + CRS_CODE)
);

-- Create a table to store assignment details for courses
CREATE TABLE ASSIGNMENT (
    ASSIGN_ID INT PRIMARY KEY,               -- Unique identifier for each assignment (Primary Key)
    ASSIGN_TITLE NVARCHAR(100),              -- Title of the assignment
    ASSIGN_DESCRIPTION NVARCHAR(255),        -- Description of the assignment
    ASSIGN_DUE_DATE DATE,                    -- Due date for the assignment
    CRS_CODE INT                             -- Foreign Key referencing the course the assignment belongs to
);

-- Create a table to store student submissions for assignments
CREATE TABLE SUBMISSION (
    STUDENT_ID INT,                          -- Foreign Key referencing the student who submitted the assignment
    ASSIGNMENT_ID INT,                       -- Foreign Key referencing the assignment submitted
    SUBM_DATE DATE,                          -- Date of submission
    SUBM_GRADE NVARCHAR(2),                  -- Grade awarded for the submission
    PRIMARY KEY (STUDENT_ID, ASSIGNMENT_ID)  -- Composite Primary Key (STUDENT_ID + ASSIGNMENT_ID)
);

-- Create a table to store learning materials for courses
CREATE TABLE LEARNING_MATERIAL (
    MATER_ID INT PRIMARY KEY,                -- Unique identifier for each learning material (Primary Key)
    MATER_TITLE NVARCHAR(100),               -- Title of the learning material
    MATER_TYPE NVARCHAR(50),                 -- Type of material (e.g., "PDF", "Video", "Slide")
    MATER_UPLOAD_DATE DATE,                  -- Date the material was uploaded
    CRS_CODE INT,                            -- Foreign Key referencing the course the material belongs to
    ACCESS_PERMISSION NVARCHAR(50)           -- Access permissions for the material (e.g., "Public", "Private")
);

-- Create a trigger named `trg_prevent_duplicate_enrollment`
-- This trigger fires BEFORE an INSERT operation on the ENROLLMENT table
CREATE TRIGGER trg_prevent_duplicate_enrollment
BEFORE INSERT ON ENROLLMENT
FOR EACH ROW
BEGIN
    -- Declare a variable to store the count of existing enrollments for the student and course
    DECLARE count_enrollments INT;

    -- Check if the student is already enrolled in the course
    SELECT COUNT(*) INTO count_enrollments
    FROM ENROLLMENT
    WHERE STUDENT_ID = NEW.STUDENT_ID AND CRS_CODE = NEW.CRS_CODE;

    -- If the student is already enrolled, raise an error
    IF count_enrollments > 0 THEN
        SIGNAL SQLSTATE '45000' -- SQLSTATE for a custom error
        SET MESSAGE_TEXT = 'Duplicate enrollment detected.'; -- Error message
    END IF;
END$$

-- Create a function named `avg_grade_course`
-- This function calculates the average grade for a given course code
CREATE FUNCTION avg_grade_course(course_code INT) RETURNS FLOAT
BEGIN
    -- Declare a variable to store the calculated average grade
    DECLARE avg_grade FLOAT;

    -- Calculate the average grade for the course
    SELECT AVG(
        CASE
            WHEN SUBM_GRADE = 'A' THEN 4.0 -- Map 'A' to 4.0
            WHEN SUBM_GRADE = 'B' THEN 3.0 -- Map 'B' to 3.0
            WHEN SUBM_GRADE = 'C' THEN 2.0 -- Map 'C' to 2.0
            WHEN SUBM_GRADE = 'D' THEN 1.0 -- Map 'D' to 1.0
            WHEN SUBM_GRADE = 'F' THEN 0.0 -- Map 'F' to 0.0
            ELSE NULL -- Ignore invalid or NULL grades
        END
    ) INTO avg_grade
    FROM SUBMISSION S
    JOIN ASSIGNMENT A ON S.ASSIGNMENT_ID = A.ASSIGN_ID
    WHERE A.CRS_CODE = course_code; -- Filter by the provided course code

    -- Return the average grade, or 0 if no grades are found (to handle NULL cases)
    RETURN IFNULL(avg_grade, 0);
END$$

-- Create a function named `total_payments` that takes a `student_id` as input
-- and returns the total amount of fees paid by the student as a FLOAT value.
CREATE FUNCTION total_payments(student_id INT) RETURNS FLOAT
BEGIN
    -- Declare a variable `total` to store the sum of the course fees.
    DECLARE total FLOAT;

    -- Calculate the total fees paid by the student by summing up the `CRS_FEES`
    -- from the `COURSE` table for all courses where the student has made a payment.
    -- The `ENROLLMENT` table is joined with the `COURSE` table using the `CRS_CODE`
    -- to get the course fees. The condition `E.PAYMENT = 'Paid'` ensures that only
    -- paid enrollments are considered.
    SELECT SUM(C.CRS_FEES) INTO total
    FROM ENROLLMENT E
    JOIN COURSE C ON E.CRS_CODE = C.CRS_CODE
    WHERE E.STUDENT_ID = student_id AND E.PAYMENT = 'Paid';

    -- Return the total fees paid by the student. If no payments were found,
    -- return 0 using the `IFNULL` function to handle cases where `total` is NULL.
    RETURN IFNULL(total, 0);
END$$

-- Create a function named `check_material_permission`
-- This function checks if a learning material has a specific access permission
CREATE FUNCTION check_material_permission(material_id INT, permission NVARCHAR(50)) RETURNS BOOLEAN
BEGIN
    -- Declare a variable to store the permission check result
    DECLARE has_permission BOOLEAN;

    -- Check if the learning material has the specified permission
    SELECT CASE 
        WHEN ACCESS_PERMISSION = permission THEN TRUE -- Return TRUE if permissions match
        ELSE FALSE -- Return FALSE if permissions do not match
    END INTO has_permission
    FROM LEARNING_MATERIAL
    WHERE MATER_ID = material_id; -- Filter by the provided material ID

    -- Return the result of the permission check
    RETURN has_permission;
END$$

-- Create a function named `days_to_deadline`
-- This function calculates the number of days remaining until the deadline of an assignment
CREATE FUNCTION days_to_deadline(assign_id INT) RETURNS INT
BEGIN
    -- Declare a variable to store the number of days left
    DECLARE days_left INT;

    -- Calculate the number of days between the assignment due date and the current date
    SELECT DATEDIFF(ASSIGN_DUE_DATE, CURDATE()) INTO days_left
    FROM ASSIGNMENT
    WHERE ASSIGN_ID = assign_id -- Filter by the provided assignment ID
    LIMIT 1; -- Ensure only one row is returned

    -- Return the number of days left, or 0 if no assignment is found (to handle NULL cases)
    RETURN IFNULL(days_left, 0);
END$$

-- Create a stored procedure named `submit_assignment`
-- This procedure allows a student to submit an assignment, ensuring no duplicate submissions
CREATE PROCEDURE submit_assignment(
    IN student_id INT,      -- Input parameter: Student ID
    IN assign_id INT,       -- Input parameter: Assignment ID
    IN subm_date DATE,      -- Input parameter: Submission date
    IN grade NVARCHAR(2)    -- Input parameter: Grade for the submission
)
BEGIN
    -- Declare a variable to store the count of existing submissions for the student and assignment
    DECLARE submission_count INT;

    -- Check if the student has already submitted THIS assignment
    SELECT COUNT(*) INTO submission_count
    FROM SUBMISSION
    WHERE STUDENT_ID = student_id AND ASSIGNMENT_ID = assign_id;

    -- If the student has NOT submitted the assignment, insert the new submission
    IF submission_count = 0 THEN
        INSERT INTO SUBMISSION (STUDENT_ID, ASSIGNMENT_ID, SUBM_DATE, SUBM_GRADE)
        VALUES (student_id, assign_id, subm_date, grade);
    ELSE
        -- If the student has already submitted the assignment, raise an error
        SIGNAL SQLSTATE '45000' -- SQLSTATE for a custom error
        SET MESSAGE_TEXT = 'Student has already submitted this assignment.'; -- Error message
    END IF;
END$$
